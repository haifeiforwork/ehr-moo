		   	plot1 = $.jqplot('chart1',[s1],{ 
		       title: 'Highlighting',  // 타이틀 설정
		       legend: {
		       		show:true,     // 범주 사용여부
		       		location: 'nw',  // 범주 위치(nw, n, ne, e, se, s, sw, w)
		       		yoffset: 6,     // offset from the inside edge of the plot in the y direction in pixels.
		       		labels:['과자값']  // 범주 이름 설정
		       },  // 범주 옵션
		       axes: {
		           xaxis: {
		               renderer: $.jqplot.DateAxisRenderer,  
		               tickOptions: {
		                   formatString: '%Y-%m'  // x축 하단 표시 설정
		               },
		               numberTicks: 4  // x축 하단 표시 갯수
		           },
		           yaxis: {
		               tickOptions: {
		                   formatString: '$%.2f'  // y 하단 표시 설정
		               }
		           }
		       },
		       highlighter: {
		           sizeAdjust: 10,
		           tooltipLocation: 'e',
		           tooltipAxes: 'y',     
		           useAxesFormatters: false,
		           tooltipFormatString: 'hello $%.2f'     // 표시 박스 설정
		       },
		       cursor: {
		           show: true
		       }
		   });
		   
		   plot2 = $.jqplot('chart2', [s2, s3, s4, s5], {
			   legend: {
				   show:true, 
				   location: 'e', 
				   placement: 'outside',
				   labels:['BBS','SMS','Q&A','MAIL']
				  },
			   highlighter: { 
				   bringSeriesToFront: true, 
				   tooltipLocation: 'e', 
				   tooltipOffset: 0, 
				   formatString: '
수치 : %s: %s
' 
				} 
		   });
		   
		   plot3 = $.jqplot('chart3', [s2, s3], { 
			   legend: {
				   show:true, 
				   location: 'nw', 
				   labels:['BBS','MAIL']
				  }
		   });
		   
		   plot4 = $.jqplot('chart4', [s2, s3, s4], { 
			   highlighter: { 
				   bringSeriesToFront: true, 
				   tooltipLocation: 'e', 
				   tooltipOffset: 0, 
				   formatString: '
Data %s: %s
' 
				} 
		   });
		   
		   line1 = [14, 32, 41, 44, 40, 47, 53, 67];

       	         xaxis:{min:3}
