(function($) {
	$(document).ready(function() {
		//core
		iKEP.debug($("#divExample")); 
		iKEP.debug($(".guideConTitle")); 
		iKEP.debug($("h4", document.body)); 
		
		// selector
		iKEP.debug($("form"));			// element selector : document.getElementsByTagName("form")
		iKEP.debug($("#divExample"));	// getElementById : document.getElementById("divExample")
		iKEP.debug($(".password"));		// class selector : document.getElementsByClassName("guideConTitle") //HTML5
		
		iKEP.debug($("input.user-id"));	// element & class selector
		iKEP.debug($("input[name=userId]"));	// attribute selector
		// fmLogin.userId.value = "abcd";	// $(document.getElementById("fmLogin").userId).val("abcd")

		
		// attributes
		iKEP.debug($("form").attr("action"));
		$("input.user-id").removeClass("text").val("user id");
		
		// traversing
		$("input.user-id").parents("tr:first").children().eq(0).text("사용자 아이디");
		iKEP.debug($("input.user-id").parents("tr").children().eq(0));
		iKEP.debug($("input.user-id").parents("tr").children()[0]);
		
		// manipulation
		$("input.password").parent().append('<span style="color:red;">*비밀번호를 입력해주세요</span>');

		$.template("empInfo", '<span>${empNo} / ${name}</span>');
		$("table tbody", "#divExample").prepend('<tr><td>사번</td><td></td></tr>');
		$("table tr:first", "#divExample").children().eq(1).append($.tmpl("empInfo", {empNo:"000", name:"이름"}));
		
		//event
		function boardSearch() {
			if (!$.trim($(this).siblings("input").val())) {
				alert("검색어를 입력해주세요.");
				return false;
			}
			
			alert("검색...");
		}
		
		$("a", "#divSearch").bind("click", boardSearch);	// function parameter
		
		$("input", "#divSearch").keypress(function(event){
			if(event.which == 13) {
				//boardSearch();
				$(this).siblings("a").trigger("click");
			}
		});
		
		$("tr", "#tblBoardList").live("click", function() {
			$(this).toggleClass("bgSelected");	// hasClass
		});
		
		// unbind, die
		// hover, toggle
		/*$("tr", "#tblBoardList").toggle(
			function() { $(this).addClass("bgSelected"); },
			function() { $(this).removeClass("bgSelected");}
		);*/
		
		
		//utility
		iKEP.debug($.trim("      trim  !    "));
		$.each([1, 2, 3], function(index) { iKEP.debug(this); });
		// $("input[type=radio]").each(function(idx) { iKEP.debug($(this)); });
		
		$.data($("#tblBoardList")[0], "currPage", 1);	// $("#tblBoardList").attr("currPage", 1);
		iKEP.debug($.data($("#tblBoardList")[0], "currPage"));
		
		$("input[type=button]").click(function() {
			var $table = $("#tblBoardList");
			var currPage = $.data($table[0], "currPage");
			switch($(this).val()) {
				case "priv" : $.data($table[0], "currPage", currPage-1);
					break;
				case "next" : $.data($table[0], "currPage", currPage+1);
					break;
			}
			iKEP.debug($.data($("#tblBoardList")[0], "currPage"));
		});
		
		// template
		$.template("tmplBoardItem", '<tr><td>${no}</td><td>${title}</td><td>${name}</td><td>${date}</td></tr>');
		$("input[value=itemAppend]").click(function() {
			var data = {no:7, title:"새로 추가한 게시물", name:"새로운 작성자", date:"오늘"};
			$("tbody", "#tblBoardList").append($.tmpl("tmplBoardItem", data));
		});
		
		//cookie
		$jq.cookie(key, value, options);
		// options : expires, path, domain, secure
	});
})(jQuery);
