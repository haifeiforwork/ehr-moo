JavaScript
		// javascript
		$(document).ready(function () {
		
			// 예제1
        	var s1 = [2, 6, 7, 10]; // 데이타
            var ticks = ['a', 'b', 'c', 'd']; // 범주
            
            var plot1 = $.jqplot('chart1', [s1], {
            	seriesDefaults:{
                    renderer:$.jqplot.BarRenderer,  // 차트 형태
                    pointLabels: { show: true }     // 포인트 표시 여부
                },
                axes: {
                    xaxis: {
                        renderer: $.jqplot.CategoryAxisRenderer,  // x축 표시
                        ticks: ticks                // x축 범주 표시
                    }
                },
                highlighter: { show: false }  // highlight 기능 비활성화
            }); 
         
            // 예제2
            var c2s1 = [2, 6, 7, 10];
            var c2s2 = [7, 5, 3, 2];
            var c2ticks = ['a', 'b', 'c', 'd'];
            var plot2 = $.jqplot('chart2', [c2s1, c2s2], {
                seriesDefaults: {
                    renderer:$.jqplot.BarRenderer,
                    pointLabels: { show: true }
                },
                axes: {
                    xaxis: {
                        renderer: $.jqplot.CategoryAxisRenderer,
                        ticks: c2ticks
                    }
                }
            });
            
            // 예제3
            var plot3 = $.jqplot('chart3', [[[2,1], [4,2], [6,3], [3,4]], [[5,1], [1,2], [3,3], [4,4]], [[4,1], [7,2], [1,3], [2,4]]], {
            	seriesDefaults: {
                    renderer:$.jqplot.BarRenderer,
                    pointLabels: { show: true, location: 'e', edgeTolerance: -15 }, // 포인트 e(왼쪽)에 표시
                    shadowAngle: 135,    // 그림자 각도
                    rendererOptions: {
                        barDirection: 'horizontal'  // bar 수평표시
                    }
                },
                axes: {
                    yaxis: {
                        renderer: $.jqplot.CategoryAxisRenderer
                    }
                }
            }); 
            
            // 예제4
            var c4s1 = [2, 6, 7, 10];
            var c4s2 = [7, 5, 3, 2];
            var c4s3 = [14, 9, 3, 8];
            
            var plot4 = $.jqplot('chart4', [c4s1, c4s2, c4s3], {
            	stackSeries: true,
                captureRightClick: true,  // 오른쪽 마우스 클릭 활성화
                seriesDefaults:{
                    renderer:$.jqplot.BarRenderer,
                    rendererOptions: {
                        highlightMouseDown: true   // 클릭 시  highlight
                    },
                    pointLabels: {show: true}   // 포인트 표시
                },
                legend: {   // 범주 표시
                    show: true,     // 표시
                    location: 'e',  // 왼쪽(동쪽) 표시
                    placement: 'outside' // 차트 박스 밖에 표시
                }
            }); 
            
       		// 예제5
            var plot5 = $.jqplot('chart5', [[[2,1], [6,2], [7,3], [10,4]], [[7,1], [5,2],[3,3],[2,4]], [[14,1], [9,2], [9,3], [8,4]]], {
            	stackSeries: true,
                captureRightClick: true,
                seriesDefaults:{
                    renderer:$.jqplot.BarRenderer,
                    shadowAngle: 135,
                    rendererOptions: {
                        barDirection: 'horizontal',
                        highlightMouseDown: true    
                    },
                    pointLabels: {show: true, formatString: '%d'}
                },
                legend: {
                    show: true,
                    location: 'e',
                    placement: 'outside'
                },
                axes: {
                    yaxis: {
                        renderer: $.jqplot.CategoryAxisRenderer
                    }
                }
            }); 
            
        });

Method & Properties

  • chart 생성 : $.jqplot(target, data, options);
  • options :

* seriesColors : ["#4bb2c5", "#c5b47f", "#EAA228", ...] 계열(단위) 색상

* stackSeries : false 예제4,5와 같이 쌓아올리는 그래프 사용여부

* title : '' 차트 타이틀
