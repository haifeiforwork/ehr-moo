		// javascript
		$(document).ready(function () {
		
			// 예제1
        	var s1 = [['Sony',7], ['Samsumg',14], ['LG',18], ['Vizio',8], ['Insignia', 4]];  // 데이타
            
            // 예제1
            var plot1 = $.jqplot('chart1', [s1], {
                grid: {  // 차트 박스  표시 옵션
                    drawBorder: false,  // 차트 박스 테두리 표시 
                    drawGridlines: false,  // 차트 안 라인 표시
                    background: '#ffffff', // 백그라운드 칼라
                    shadow:false           // 차트 박스 그림자 표시
                },
                seriesDefaults:{
                    renderer:$.jqplot.PieRenderer,  // 차트 형태
                    rendererOptions: {             
                        showDataLabels: true        // 데이타 표시 여부
                    }
                },
                legend: {
                    show: true,          // 범주 표시 여부
                    rendererOptions: {  
                        numberRows: 1    // 표시 Rows 수
                    },
                    location: 's'        // 범주 표시 위치(nw, n, ne, e, se, s, sw, w)
                }
            }); 
            
            // 예제2
            var plot2 = $.jqplot('chart2', [s1], {
                grid: {
                    drawBorder: true, 
                    drawGridlines: false,
                    background: '#ffffff',
                    shadow:true
                },
                title:'Poll Title',   // 차트 타이틀
                seriesDefaults:{
                    renderer:$.jqplot.PieRenderer,
                    rendererOptions: {
                        sliceMargin:4   // 파이 조각 표시 시 마진값
                    }
                },
                legend: {
                    show: true,
                    location: 'e'
                }
            }); 
            
            // 예제3
            var plot3 = $.jqplot('chart3', [s1], {
                grid: {
                    drawBorder: true, 
                    drawGridlines: false,
                    background: '#ffffff',
                    shadow:true
                },                
                seriesDefaults:{
                    renderer:$.jqplot.PieRenderer,
                    shadow:false,
                    rendererOptions: {                        
                        sliceMargin:3
                    }
                },
                legend: {
                    show: true,
                    location: 'e',
                    placement: 'outside'  // 범주 차트 박스 밖에 표시
                }
            }); 
                        
            // 예제4
            var s2 = [['LG',30], ['Sony',5], ['Samsumg',6] ];
            var plot4 = $.jqplot('chart4', [s2], {
                grid: {
                    drawBorder: true, 
                    drawGridlines: false,
                    background: '#ffffff',
                    shadow:true
                },                
                seriesDefaults:{
                    renderer:$.jqplot.PieRenderer,                   
                    shadow:false,
                    rendererOptions: {
                    	showDataLabels: true,
                        sliceMargin:5,
                        startAngle: -90  
                    }
                },
                legend: {
                    show: true,
                    rendererOptions: {
                        numberRows: 1
                    },
                    location: 's'
                }
            }); 
            
        });
